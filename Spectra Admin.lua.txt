local function CreateDetectionTag(player)
    local function addTag(character)
        local head = character:WaitForChild("Head", 5)
        if not head then return end
        
        local tag = Instance.new("ObjectValue")
        tag.Name = "SpectraClient"
        tag.Value = true
        tag.Parent = head
        
        local gui = Instance.new("BillboardGui")
        gui.Name = "SpectraIndicator"
        gui.Parent = head
        gui.Size = UDim2.new(0, 50, 0, 50)
        gui.StudsOffset = Vector3.new(0, 3, 0)
        gui.AlwaysOnTop = true
        gui.Enabled = true
        
        local frame = Instance.new("Frame")
        frame.Parent = gui
        frame.Size = UDim2.new(1, 0, 1, 0)
        frame.BackgroundTransparency = 1
        
        local image = Instance.new("ImageLabel")
        image.Parent = frame
        image.Size = UDim2.new(1, 0, 1, 0)
        image.BackgroundTransparency = 1
        image.Image = "rbxassetid://77070387164904"
        image.ImageColor3 = Color3.fromRGB(255, 50, 50)
        image.ImageTransparency = 0.2
        
        coroutine.wrap(function()
            while gui and gui.Parent do
                image.Rotation = image.Rotation + 2
                task.wait(0.03)
            end
        end)()
    end
    
    if player.Character then
        addTag(player.Character)
    end
    
    player.CharacterAdded:Connect(addTag)
end

local function ScanForSpectraClients()
    SpectraClients = {}
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            if player.Character and player.Character:FindFirstChild("Head") then
                local head = player.Character.Head
                if head:FindFirstChild("SpectraClient") then
                    SpectraClients[player.UserId] = player
                end
            end
        end
    end
    
    return SpectraClients
end

local function GetScriptUsers()
    ScanForSpectraClients()
    local list = {}
    for userId, player in pairs(SpectraClients) do
        table.insert(list, player.Name .. " (" .. (player.DisplayName or player.Name) .. ")")
    end
    return list
end

local function SendCommandToClient(targetPlayer, command)
    if not targetPlayer then return false end
    
    local commandMsg = string.format("EXEC SPECTRA:%s:%s", command, targetPlayer.Name)
    
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local channel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if channel then
            channel:SendAsync(commandMsg)
        end
    else
        ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents") and
        ReplicatedStorage.DefaultChatSystemChatEvents.SayMessageRequest:FireServer(commandMsg, "All")
    end
    
    return true
end

local function Notify(title, message, duration)
    WindUI:Notification({
        Title = title,
        Content = message,
        Duration = duration or 3,
        Icon = "info"
    })
end

local function UpdatePlayerList(dropdown)
    local users = GetScriptUsers()
    if #users == 0 then
        table.insert(users, "⚠️ Nenhum usuário ativo")
    end
    if dropdown then
        dropdown:Refresh(users)
    end
    return users
end

end)()

            SpectraClients[player.UserId] = player

        end

end)

if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
    TextChatService.OnIncomingMessage = function(message)
        local speaker = message.TextSource and Players:GetPlayerByUserId(message.TextSource.UserId)
        if not speaker or not message.Text then return end
        
        local msg = message.Text
        
        if msg:match("SPECTRA CONFIRM:") then
            local cmd = msg:gsub("SPECTRA CONFIRM:", "")
            Notify("Comando Executado", speaker.Name + " executou: " + cmd, 2)
        end
        
        if msg:match("SPECTRA TAG APPLIED") then
            Notify("🏷️ Tag Ativada", speaker.Name + " agora tem a tag visual", 3)
        end
    end
end
